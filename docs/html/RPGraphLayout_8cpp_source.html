227  void GraphLayout::writeToPNG(const int image_w, const int image_h,
228  std::string path)
229  {
230  const float xRange = getXRange();
231  const float yRange = getYRange();
232  const RPGraph::Coordinate center = getCenter(); // Why do we need the center?
233  const float xCenter = center.x; // Probably to set the origin for layout.
234  const float yCenter = center.y;
235  const float minX = xCenter - xRange/2.0;
236  const float minY = yCenter - yRange/2.0;
237  const float xScale = image_w/xRange;
238  const float yScale = image_h/yRange;
239 
240  // Here we need to do some guessing as to what the optimal
241  // opacity of nodes and edges might be, given network size.
245  const float node_opacity = 10000.0 / graph.num_nodes();
246  const float edge_opacity = 100000.0 / graph.num_edges();
247 
248  // Write to file.
249  pngwriter layout_png(image_w, image_h, 0, path.c_str());
250  layout_png.invert(); // set bg. to white
251 
252  for (nid_t n1 = 0; n1 < graph.num_nodes(); ++n1)
253  {
254  // Plot node,
255  layout_png.filledcircle_blend((getX(n1) - minX)*xScale,
256  (getY(n1) - minY)*yScale,
257  3, node_opacity, 0, 0, 0);
258  for (nid_t n2 : graph.neighbors_with_geq_id(n1)) {
259  // ... and edge.
260  layout_png.line_blend((getX(n1) - minX)*xScale, (getY(n1) - minY)*yScale,
261  (getX(n2) - minX)*xScale, (getY(n2) - minY)*yScale,
262  edge_opacity, 0, 0, 0);
263  }
264  }
265  // Write it to disk.
266  layout_png.write_png();
267  }
