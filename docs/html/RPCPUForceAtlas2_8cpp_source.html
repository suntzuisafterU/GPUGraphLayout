61  void CPUForceAtlas2::apply_attract(nid_t n)
62  {
63  Real2DVector f = Real2DVector(0.0, 0.0);
64  for (nid_t t : layout.graph.neighbors_with_geq_id(n))
65  {
66  // Here we define the magnitude of the attractive force `f_a'
67  // *divided* by the length distance between `n' and `t', i.e. `f_a_over_d'
68  float f_a_over_d;
69  if (use_linlog)
70  {
71  float dist = layout.getDistance(n, t);
72  f_a_over_d = dist == 0.0 ? std::numeric_limits<float>::max() : logf(1+dist) / dist;
73  }
74 
75  else
76  {
77  f_a_over_d = 1.0;
78  }
79 
80  f += layout.getDistanceVector(n, t) * f_a_over_d;
81 
82  //TODO: this is temporary, but required due to
83  // iteration over neighbors_with_geq_id
84  forces[t] += layout.getDistanceVector(n, t) * (-f_a_over_d);
85 
86  // forces[n] += getNormalizedDistanceVector(n, t) * f_a(n, t);
87  }
88  forces[n] += f;
89  }

91  void CPUForceAtlas2::apply_repulsion(nid_t n)
92  {
93  // Approximation, O(log n), dependent on theta.
94  if (use_barneshut)
95  {
96  forces[n] += (BH_Approximator.approximateForce(layout.getCoordinate(n), mass(n), theta) * k_r);
97  }
98 
99  // Exact, O(n) (n^2 to calculate for all nodes)
100  else
101  {
102  for (nid_t t = 0; t < layout.graph.num_nodes(); ++t)
103  {
104  if (n == t) continue;
105  float distance = layout.getDistance(n, t);
106  float f_r = distance == 0.0 ? std::numeric_limits<float>::max() : k_r * mass(n) * mass(t) / distance / distance;
107  forces[n] += layout.getDistanceVector(n, t) * f_r;
108  }
109  }
110  }

112  void CPUForceAtlas2::apply_gravity(nid_t n)
113  {
114  float f_g, d;
115 
116  // `d' is the distance from `n' to the center (0.0, 0.0)
117  d = std::sqrt(layout.getX(n)*layout.getX(n) + layout.getY(n)*layout.getY(n));
118  if(d == 0.0) return;
119 
120  // Here we define the magnitude of the gravitational force `f_g'.
121  if (strong_gravity)
122  {
123  f_g = k_g*mass(n);
124  }
125 
126  else
127  {
128  f_g = k_g*mass(n) / d;
129  }
130 
131  forces[n] += (Real2DVector(-layout.getX(n), -layout.getY(n)) * f_g);
132  }

162  void CPUForceAtlas2::updateSpeeds()
163  {
164  // The following speed-update procedure for ForceAtlas2 follows
165  // the one by Gephi:
166  // https://github.com/gephi/gephi/blob/6efb108718fa67d1055160f3a18b63edb4ca7be2/modules/LayoutPlugin/src/main/java/org/gephi/layout/plugin/forceAtlas2/ForceAtlas2.java
167 
168  // `Auto adjust speeds'
169  float total_swinging = 0.0;
170  float total_effective_traction = 0.0;
171  for (nid_t nid = 0; nid < layout.graph.num_nodes(); ++nid)
172  {
173  total_swinging += mass(nid) * swg(nid); // Eq. (11)
174  total_effective_traction += mass(nid) * tra(nid); // Eq. (13)
175  }
176 
177  // We want to find the right jitter tollerance for this graph,
178  // such that totalSwinging < tolerance * totalEffectiveTraction
183  float estimated_optimal_jitter_tollerance = 0.05 * std::sqrt(layout.graph.num_nodes());
184  float minJT = std::sqrt(estimated_optimal_jitter_tollerance);
185  float jt = jitter_tolerance * fmaxf(minJT,
186  fminf(k_s_max,
187  estimated_optimal_jitter_tollerance * total_effective_traction / powf(layout.graph.num_nodes(), 2.0)
188  )
189  );
190  float min_speed_efficiency = 0.05;
191 
192  // `Protect against erratic behavior'
193  if (total_swinging / total_effective_traction > 2.0)
194  {
195  if (speed_efficiency > min_speed_efficiency) speed_efficiency *= 0.5;
196  jt = fmaxf(jt, jitter_tolerance);
197  }
198 
199  // `Speed efficiency is how the speed really corrosponds to the swinging vs. convergence tradeoff.'
200  // `We adjust it slowly and carefully'
201  float targetSpeed = jt * speed_efficiency * total_effective_traction / total_swinging;
202 
203  if (total_swinging > jt * total_effective_traction)
204  {
205  if (speed_efficiency > min_speed_efficiency)
206  {
207  speed_efficiency *= 0.7;
208  }
209  }
210  else if (global_speed < 1000)
211  {
212  speed_efficiency *= 1.3;
213  }
214 
215  // `But the speed shouldn't rise much too quickly, ... would make convergence drop dramatically'.
216  float max_rise = 0.5;
217  global_speed += fminf(targetSpeed - global_speed, max_rise * global_speed);
218  }

247  void CPUForceAtlas2::doStep()
248  {
249  if (use_barneshut) rebuild_bh();
250 
251  for (nid_t n = 0; n < layout.graph.num_nodes(); ++n)
252  {
253  apply_gravity(n);
254  apply_attract(n);
255  apply_repulsion(n);
256  }
257 
258  updateSpeeds();
259 
260  for (nid_t n = 0; n < layout.graph.num_nodes(); ++n)
261  {
262  apply_displacement(n);
263  prev_forces[n] = forces[n];
264  forces[n] = Real2DVector(0.0f, 0.0f);
265  }
266  iteration++;
267  }
