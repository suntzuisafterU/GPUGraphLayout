40  UGraph::UGraph(std::string edgelist_path)
41  {
42  node_count = 0;
43  edge_count = 0;
44 
45  std::fstream edgelist_file(edgelist_path, std::ifstream::in);
46 
47  std::string line;
48  while(std::getline(edgelist_file, line))
49  {
50  // Skip any comments
51  if(line[0] == '#') continue;
52 
53  // Read source and target from file
54  nid_t s, t;
55  std::istringstream(line) >> s >> t;
56 
57  if(s != t and !has_edge(s, t)) add_edge(s, t);
58  }
59 
60  edgelist_file.close();
61  }

68  bool UGraph::has_edge(nid_t s, nid_t t)
69  {
70  if(!has_node(s) or !has_node(t)) return false;
71 
72  nid_t s_mapped = node_map[s];
73  nid_t t_mapped = node_map[t];
74 
75  if(adjacency_list.count(std::min(s_mapped, t_mapped)) == 0) return false;
76 
77  std::vector<nid_t> neighbors = adjacency_list[std::min(s_mapped, t_mapped)];
78  if(std::find(neighbors.begin(), neighbors.end(), std::max(s_mapped, t_mapped)) == neighbors.end())
79  return false;
80  else
81  return true;
82  }

94  void UGraph::add_edge(nid_t s, nid_t t)
95  {
96  if(has_edge(s, t)) return;
97  if(!has_node(s)) add_node(s);
98  if(!has_node(t)) add_node(t);
99  nid_t s_mapped = node_map[s];
100  nid_t t_mapped = node_map[t];
101 
102  // Insert edge into adjacency_list
103  adjacency_list[std::min(s_mapped, t_mapped)].push_back(std::max(s_mapped, t_mapped));
104  degrees[s_mapped] += 1;
105  degrees[t_mapped] += 1;
106  edge_count++;
107  }

140 // CSRUGraph represents an undirected graph using a
141 // compressed sparse row (CSR) datastructure.
145  CSRUGraph::CSRUGraph(nid_t num_nodes, nid_t num_edges)
146  {
147  // `edges' is a concatenation of all edgelists (i.e. flattened edge list)
148  // `offsets' contains offset (in `edges`) for each nodes' edgelist. (index into flattened edge list)
149  // `nid_to_offset` maps nid to index to be used in `offset' (index into array of mappings id -> offset, then use offset to index into flattened array of edges.)
150 
151  // e.g. the edgelist of node with id `nid' starts at
152  // edges[offsets[nid_to_offset[nid]]] and ends at edges[offsets[nid_to_offset[nid]] + 1]
153  // (left bound inclusive right bound exclusive) (normal indexing of sublists)
154 
155  edge_count = num_edges; // num_edges counts each bi-directional edge once.
156  node_count = num_nodes;
157  edges = (nid_t *) malloc(sizeof(nid_t) * 2 * edge_count); // flattened array
158  offsets = (nid_t *) malloc(sizeof(nid_t) * node_count); // map
159  offset_to_nid = (nid_t *) malloc(sizeof(nid_t) * node_count);
160 
161  // Create a map from original ids to ids used throughout CSRUGraph
162  // std::unordered_map<nid_t, nid_t> // secret hidden header variable!
163  nid_to_offset.reserve(node_count); // IMPORTANT: reserves space for at least n elements (dynamic datastructure.)
164 
165  first_free_id = 0;
166  edges_seen = 0;
167  }
168 
169  CSRUGraph::~CSRUGraph()
170  {
171  free(edges);
172  free(offsets);
173  free(offset_to_nid);
174  }

176  void CSRUGraph::insert_node(nid_t node_id, std::vector<nid_t> nbr_ids)
177  {
178  nid_t source_id_old = node_id;
179  nid_t source_id_new = first_free_id;
180  nid_to_offset[source_id_old] = first_free_id;
181  offset_to_nid[first_free_id] = source_id_old;
182  first_free_id++;
183 
184  offsets[source_id_new] = edges_seen;
185  for (auto nbr_id : nbr_ids)
186  {
187  nid_t dest_id_old = nbr_id;
188  edges[edges_seen] = dest_id_old;
189  edges_seen++;
190  }
191  }

201  nid_t CSRUGraph::degree(nid_t nid)
202  {
203  // If nid is last element of `offsets'... we prevent out of bounds.
204  nid_t r_bound;
205  if (nid < node_count - 1) r_bound = offsets[nid+1];
206  else r_bound = edge_count * 2;
207  nid_t l_bound = offsets[nid];
208  return (r_bound - l_bound);
209  }
