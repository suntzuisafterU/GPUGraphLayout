54  void UGraph::read_edgelist_file(std::string edgelist_path) {
55  std::fstream edgelist_file(edgelist_path, std::ifstream::in);
56 
57  std::string line;

59  {
60  // Skip any comments

65  std::istringstream(line) >> s >> t;

111  void UGraph::add_edge(nid_t s, nid_t t)
112  {
113  if(has_edge(s, t) or s == t) return;
114  if(!has_node(s)) add_node(s);
115  if(!has_node(t)) add_node(t);
116  nid_t s_mapped = node_map[s];
117  nid_t t_mapped = node_map[t];
118 
119  // Insert edge into adjacency_list
120  adjacency_list[std::min(s_mapped, t_mapped)].push_back(std::max(s_mapped, t_mapped));
121  degrees[s_mapped] += 1;
122  degrees[t_mapped] += 1;
123  edge_count++;
124  }

164 // CSRUGraph represents an undirected graph using a
165 // compressed sparse row (CSR) datastructure.
169  CSRUGraph::CSRUGraph(nid_t num_nodes, nid_t num_edges)
170  {
171  // `edges' is a concatenation of all edgelists (i.e. flattened edge list)
172  // `offsets' contains offset (in `edges`) for each nodes' edgelist. (index into flattened edge list)
173  // `nid_to_offset` maps nid to index to be used in `offset' (index into array of mappings id -> offset, then use offset to index into flattened array of edges.)
174 
175  // e.g. the edgelist of node with id `nid' starts at
176  // edges[offsets[nid_to_offset[nid]]] and ends at edges[offsets[nid_to_offset[nid]] + 1]
177  // (left bound inclusive right bound exclusive) (normal indexing of sublists)
178 
179  edge_count = num_edges; // num_edges counts each bi-directional edge once.
180  node_count = num_nodes;
181  edges = (nid_t *) malloc(sizeof(nid_t) * 2 * edge_count); // flattened array
182  offsets = (nid_t *) malloc(sizeof(nid_t) * node_count); // map
183  offset_to_nid = (nid_t *) malloc(sizeof(nid_t) * node_count);
184 
185  // Create a map from original ids to ids used throughout CSRUGraph
186  // std::unordered_map<nid_t, nid_t> // secret hidden header variable!
187  nid_to_offset.reserve(node_count); // IMPORTANT: reserves space for at least n elements (dynamic datastructure.)
188 
189  first_free_id = 0;
190  edges_seen = 0;
191  }
192 
193  CSRUGraph::~CSRUGraph()
194  {
195  free(edges);
196  free(offsets);
197  free(offset_to_nid);
198  }

200  void CSRUGraph::insert_node(nid_t node_id, std::vector<nid_t> nbr_ids)
201  {
202  nid_t source_id_old = node_id;
203  nid_t source_id_new = first_free_id;
204  nid_to_offset[source_id_old] = first_free_id;
205  offset_to_nid[first_free_id] = source_id_old;
206  first_free_id++;
207 
208  offsets[source_id_new] = edges_seen;
209  for (auto nbr_id : nbr_ids)
210  {
211  nid_t dest_id_old = nbr_id;
212  edges[edges_seen] = dest_id_old;
213  edges_seen++;
214  }
215  }

225  nid_t CSRUGraph::degree(nid_t nid)
226  {
227  // If nid is last element of `offsets'... we prevent out of bounds.
228  nid_t r_bound;
229  if (nid < node_count - 1) r_bound = offsets[nid+1];
230  else r_bound = edge_count * 2;
231  nid_t l_bound = offsets[nid];
232  return (r_bound - l_bound);
233  }
