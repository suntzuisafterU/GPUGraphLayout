47  void BarnesHutCell::add_leafcell(int quadrant, float mass, Coordinate pos)
48  {
49  Coordinate leafcell_center_coordinate = Coordinate(0,0);
50  if (quadrant == 0)
51  leafcell_center_coordinate = Coordinate(this->cell_center.x-length/4.0,this->cell_center.y+length/4);
52  else if (quadrant == 1)
53  leafcell_center_coordinate = Coordinate(this->cell_center.x+length/4.0,this->cell_center.y+length/4);
54  else if (quadrant == 2)
55  leafcell_center_coordinate = Coordinate(this->cell_center.x+length/4.0,this->cell_center.y-length/4);
56  else if (quadrant == 3)
57  leafcell_center_coordinate = Coordinate(this->cell_center.x-length/4.0,this->cell_center.y-length/4);
58 
59  sub_cells[quadrant] = new BarnesHutCell(leafcell_center_coordinate, this->length/2.0, pos, mass);
60  num_subparticles += 1;
61 
62  }

86  Real2DVector BarnesHutApproximator::approximateForce(Coordinate particle_pos, float particle_mass, float theta)
87  {
88  Real2DVector force = Real2DVector(0.0, 0.0);
89  std::queue<BarnesHutCell*> cells_to_check;
90  cells_to_check.push(root_cell);
91 
92  BarnesHutCell *cur_cell;
93  while (!cells_to_check.empty())
94  {
95  cur_cell = cells_to_check.front();
96  cells_to_check.pop();
97 
98  // TODO: Why is D2 const?
99  const float D2 = distance2(particle_pos, cur_cell->mass_center);
100  if (D2 == 0)
101  {
102  // If we approximate the force of a particle on itself...
103  if (cur_cell->num_subparticles == 0) continue;
104  else return Real2DVector(rand(), rand());
105 
106  }
107 
108  // length / D >= theta is the criterion to divide into subcells.
109  if (cur_cell->length*cur_cell->length / D2 < theta*theta || cur_cell->num_subparticles == 0)
110  force += direction(particle_pos, cur_cell->mass_center) *
111  (particle_mass * cur_cell->total_mass / D2);
112 
113  else
114  for (int i = 0; i < 4; ++i)
115  if (cur_cell->sub_cells[i] != nullptr) cells_to_check.push(cur_cell->sub_cells[i]);
116  }
117  return force;
118  }

120  void BarnesHutApproximator::insertParticle(RPGraph::Coordinate particle_position, float particle_mass)
121  {
122  if(not root_cell)
123  {
124  root_cell = new BarnesHutCell(this->root_center, this->root_length,
125  particle_position, particle_mass);
126  }
127 
128  else
129  {
130  BarnesHutCell *cur_cell = root_cell;
131  while (true)
132  {
133  const int quadrant_new_particle = (particle_position-cur_cell->cell_center).quadrant();
134 
135  if (particle_position.y > cur_cell->ub or particle_position.x > cur_cell->rb or
136  particle_position.x < cur_cell->lb or particle_position.y < cur_cell->bb)
137  {
138  //fprintf(stderr, "error: Barnes-Hut: Can't insert particle out of bounds of this cell.\n");
139  return;
140  }
141 
142  // N.B. a BarnesHutCell is never empty, but can lack subparticles/cells.
143  // If so, we need to create, and insert, a subcell for the single particle that
144  // is stored in this cell.
145  if (cur_cell->num_subparticles == 0)
146  {
147  if (particle_position == cur_cell->mass_center)
148  {
149  // We want two particles in the same place...
150  // Thats equivalent to a single particle with summed masses.
151  // mass_center won't change.
152  cur_cell->total_mass += particle_mass;
153  return;
154  }
155 
156  // We move the single particle to a subcell.
157  int quadrant_existing_particle = (cur_cell->mass_center - cur_cell->cell_center).quadrant();
158  cur_cell->add_leafcell(quadrant_existing_particle, cur_cell->total_mass, cur_cell->mass_center);
159  }
160 
161  // We assume inserting will succeed, and update total_mass and mass_center accordingly
162  cur_cell->total_mass += particle_mass;
163  cur_cell->mass_center = cur_cell->mass_center * (float) (cur_cell->num_subparticles);
164  cur_cell->mass_center += particle_position;
165  cur_cell->mass_center /= (cur_cell->num_subparticles+1);
166 
167  // If we can add a leaf-cell in an empty slot, we do so.
168  if (cur_cell->sub_cells[quadrant_new_particle] == nullptr)
169  {
170  cur_cell->add_leafcell(quadrant_new_particle, particle_mass, particle_position);
171  return;
172  }
173 
174  // Else we recurse to the occupied cell.
175  else
176  {
177  cur_cell->num_subparticles += 1;
178  cur_cell = cur_cell->sub_cells[quadrant_new_particle];
179  }
180  }
181  }
182  }
